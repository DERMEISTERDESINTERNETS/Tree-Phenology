8.4 Code

My code for the whole chapter:

Code 
require(chillR)
require(ggplot2)
require(tidyr)
require(kableExtra)

# Task1

Days <- daylength(latitude = 46.62, JDay = 1:365)
Days_df <-
  data.frame(
    JDay = 1:365,
    Sunrise = Days$Sunrise,
    Sunset = Days$Sunset,
    Daylength = Days$Daylength
  )
Days_df <- pivot_longer(Days_df,cols=c(Sunrise:Daylength))

ggplot(Days_df, aes(JDay, value)) +
  geom_line(lwd = 1.5) +
  facet_grid(cols = vars(name)) +
  ylab("Time of Day / Daylength (Hours)") +
  theme_bw(base_size = 20) +
  scale_x_continuous(expand = c(0,0)) +
  ggtitle("Data for Klagenfurt, Austria (Latitude of 46.62)")
