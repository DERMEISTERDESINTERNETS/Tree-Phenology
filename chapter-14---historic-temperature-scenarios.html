13 Chapter 14 - Historic temperature scenarios

13.1 Questions

  1. For the location you chose for previous exercises, produce historic temperature scenarios representing several years of the historic record (your choice).

  2. Produce chill distributions for these scenarios and plot them.

13.2 Answers

1. Finding the location for extracting then computing and summarizing the data:

library(chillR)
library(dplyr)
library(ggplot2)

# Loading Temperature data from previous lessons

Klagenfurt_temps <- read.csv2("data/Klagenfurt_temps.csv")

scenario_1980 <- temperature_scenario_from_records(weather = Klagenfurt_temps,
                                                   year = 1980)



scenario_1980$'1980'$data


temps_1980 <- temperature_generation(weather = Klagenfurt_temps,
                                     years = c(1973,2019),
                                     sim_years = c(2001,2100),
                                     temperature_scenario = scenario_1980)




#adjust baseline

scenario_1996 <- temperature_scenario_from_records(weather = Klagenfurt_temps,
                                                   year = 1996)

# Relative scenario

relative_scenario <- temperature_scenario_baseline_adjustment(
  baseline = scenario_1996,
  temperature_scenario = scenario_1980)

# apply the relative scenario

temps_1980<-temperature_generation(weather = Klagenfurt_temps,
                                   years = c(1973,2019),
                                   sim_years = c(2001,2100),
                                   temperature_scenario = relative_scenario)


# generating all past scenarios

all_past_scenarios <- temperature_scenario_from_records(
  weather = Klagenfurt_temps,
  year = c(1980,
           1990,
           2000,
           2010))

adjusted_scenarios <- temperature_scenario_baseline_adjustment(
  baseline = scenario_1996,
  temperature_scenario = all_past_scenarios)

all_past_scenario_temps <- temperature_generation(
  weather = Klagenfurt_temps,
  years = c(1973,2019),
  sim_years = c(2001,2100),
  temperature_scenario = adjusted_scenarios)

save_temperature_scenarios(all_past_scenario_temps, "data", "Klagenfurt_hist_scenarios_chapter14")

# Frost model function

frost_model <- function(x)
  step_model(x,
             data.frame(
               lower=c(-1000,0),
               upper=c(0,1000),
               weight=c(1,0)))

models <- list(Chill_Portions = Dynamic_Model,
               GDH = GDH,
               Frost_H = frost_model)

chill_hist_scenario_list <- tempResponse_daily_list(all_past_scenario_temps,
                                                    latitude = 46.62,
                                                    Start_JDay = 305,
                                                    End_JDay = 59,
                                                    models = models)

#saving the data

chill_hist_scenario_list <- lapply(chill_hist_scenario_list,
                                   function(x) x %>%
                                     filter(Perc_complete == 100))

save_temperature_scenarios(chill_hist_scenario_list, "data","Klagenfurt_hist_chill_305_59_Chapter_14")

# Lets sort the data and make a nice plot out of it

scenarios <- names(chill_hist_scenario_list)[1:4]

all_scenarios <- chill_hist_scenario_list[[scenarios[1]]] %>%
  mutate(scenario = as.numeric(scenarios[1]))

for (sc in scenarios[2:4])
  all_scenarios <- all_scenarios %>%
  rbind(chill_hist_scenario_list[[sc]] %>%
          cbind(
            scenario=as.numeric(sc))
  ) %>%
  filter(Perc_complete == 100)

# Let's compute the actual 'observed' chill for comparison
actual_chill <- tempResponse_daily_list(Klagenfurt_temps,
                                        latitude=46.62,
                                        Start_JDay = 305,
                                        End_JDay = 59,
                                        models)[[1]] %>%
  filter(Perc_complete == 100)

ggplot(data = all_scenarios,
       aes(scenario,
           Chill_Portions,
           fill = factor(scenario))) +
  geom_violin() +
  ylab("Chill accumulation (Chill Portions)") +
  xlab("Scenario year") +
  theme_bw(base_size = 15) +
  ylim(c(0,90)) +
  geom_point(data = actual_chill,
             aes(End_year,
                 Chill_Portions,
                 fill = "blue"),
             col = "blue",
             show.legend = FALSE) +
  scale_fill_discrete(name = "Scenario",
                      breaks = unique(all_scenarios$scenario))
