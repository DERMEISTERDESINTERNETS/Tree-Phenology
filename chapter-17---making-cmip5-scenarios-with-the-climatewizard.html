library(chillR)
library(tidyverse)
# Download

getClimateWizardData(
  coordinates = c(longitude = 14.31,
                  latitude = 46.62),
  scenario = "rcp45",
  start_year = 2020,
  end_year = 2050,
  metric = c("CD18", "R02"),
  GCMs = c("bcc-csm1-1", "BNU-ESM")
)

# Processing

RCPs <- c("rcp45",
          "rcp85")
Times <- c(2050,
           2085)

for(RCP in RCPs)
  for(Time in Times)
  {start_year <- Time - 15
  end_year <- Time + 15
  clim_scen <-
    getClimateWizardData(
      c(longitude = 14.31,
        latitude = 46.62),
      RCP,
      start_year,
      end_year,
      temperature_generation_scenarios = TRUE,
      baseline =c(1975, 2005),
      metric = "monthly_min_max_temps",
      GCMs = "all")
  save_temperature_scenarios(clim_scen,
                             "data/ClimateWizard",
                             paste0("klagenfurt_futures_",Time,"_",RCP))}

# baseline adjustment

klagenfurt_temps<- read.csv2("data/Klagenfurt_temps.csv")[
  , c("Year", "Month", "Day", "Tmax", "Tmin")
]

scenario_1990 <- klagenfurt_temps %>%
  temperature_scenario_from_records(1990)
scenario_1996 <- klagenfurt_temps %>%
  temperature_scenario_from_records(1996)
adjustment_scenario <-
  temperature_scenario_baseline_adjustment(scenario_1996,
                                           scenario_1990)

print(adjustment_scenario)

# Selecting RCPs

RCPs <- c("rcp45",
          "rcp85")
Times <- c(2050,
           2085)

# Saving generated weather

for(RCP in RCPs)
  for(Time in Times)
  {
    clim_scen <- load_ClimateWizard_scenarios(
      "data/ClimateWizard",
      paste0("klagenfurt_futures_",
             Time,
             "_",
             RCP))
    
    clim_scen_adjusted <-
      temperature_scenario_baseline_adjustment(
        baseline_temperature_scenario = adjustment_scenario,
        temperature_scenario = clim_scen)
    
    Temps <- temperature_generation(
      weather = klagenfurt_temps, 
      years = c(1973,
                2019),
      sim_years = c(2001,
                    2101),
      temperature_scenario = clim_scen_adjusted)
    
    save_temperature_scenarios(
      Temps,
      "data/Weather_ClimateWizard",
      paste0("klagenfurt_",
             Time,
             "_",
             RCP))
  }

# Next

all_past_scenarios <- temperature_scenario_from_records(
  weather = klagenfurt_temps,
  year = c(1980,
           1990,
           2000,
           2010))

adjusted_scenarios <- temperature_scenario_baseline_adjustment(
  baseline = scenario_1996,
  temperature_scenario = all_past_scenarios)

all_past_scenario_temps <- temperature_generation(
  weather = klagenfurt_temps,
  years = c(1973,
            2019),
  sim_years = c(2001,
                2101),
  temperature_scenario = adjusted_scenarios)

save_temperature_scenarios(

  "data/Weather_ClimateWizard",
  "klagenfurt_historic")

# Frost Model

frost_model <- function(x)
  step_model(x,
             data.frame(
               lower=c(-1000,0),
               upper=c(0,1000),
               weight=c(1,0)))

models <- list(Chill_Portions = Dynamic_Model,
               GDH = GDH,
               Frost_H = frost_model)

# Apply to historic data

Temps <- load_temperature_scenarios("data/Weather_ClimateWizard",
                                    "klagenfurt_historic")

chill_past_scenarios <-
  Temps %>%

    latitude = 46.62,
    Start_JDay = 305,
    End_JDay = 59,
    models = models,
    misstolerance = 10)
chill_observed <-

save_temperature_scenarios(chill_past_scenarios,

    chill <- load_temperature_scenarios(

    if(RCP == "rcp45") RCPcaption <- "RCP4.5"
    if(RCP == "rcp85") RCPcaption <- "RCP8.5"
    if(Time == "2050") Time_caption <- "2050"
    if(Time == "2085") Time_caption <- "2085"
    chills <- chill %>% 
      make_climate_scenario(
        caption = c(RCPcaption,

        add_to = chills)
